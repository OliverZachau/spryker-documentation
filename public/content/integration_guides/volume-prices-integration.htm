Volume Prices Feature Integration

The Volume Prices Feature is shipped with the following modules:

>
Module Description
PriceProduct Responsible for extraction of prices from price data and resolving prices in Zed.
PriceProductStorageResponsible for extraction of prices from price data and resolving prices in Yves.
PriceProductVolume Provides functionality to extract and filter volume prices.
PriceProductVolumeWidget Provides widget to render volume prices for product.

To install the Volume Pricing feature, follow the steps below.

  1. Install the necessary modules using composer:

    Update the existing and install the required modules:

    composer update "spryker/*" "spryker-shop/*"
    composer require spryker/price-product-volume:"^1.0.0" spryker-shop/price-product-volume-widget:"^1.0.0" --update-with-dependencies
  2. Add plugins to Client's PriceProductStorageDependencyProvider
    ModulePluginDescriptionMethod in Dependency Provider
    PriceProductVolumePriceProductVolumeExtractorPluginExtracts volume prices from storage price data.getPriceProductPricesExtractorPlugins()
    
	<?php
	namespace Pyz\Client\PriceProductStorageDependencyProvider;
 
	use Spryker\Client\PriceProductVolume\Plugin\PriceProductStorageExtension\PriceProductVolumeExtractorPlugin;
  
	class PriceProductStorageDependencyProvider extends SprykerPriceProductStorageDependencyProvider
	{
		/**
		 * @return \Spryker\Client\PriceProductStorageExtension\Dependency\Plugin\PriceProductStoragePricesExtractorPluginInterface[]
		 */
		protected function getPriceProductPricesExtractorPlugins(): array
		{
			return [
				new PriceProductVolumeExtractorPlugin(),
			];
		}
	}
  
						}
  3. Add plugins to Service's PriceProductDependencyProvider
    ModulePluginDescriptionMethod in Dependency Provider
    PriceProductVolumePriceProductVolumeFilterPluginFilters volume prices from prices collection.getPriceProductDecisionPlugins()
    
	<?php
	namespace Pyz\Service\PriceProduct;
  
	use Spryker\Service\PriceProductVolume\Plugin\PriceProductExtension\PriceProductVolumeFilterPlugin;
   
	class PriceProductDependencyProvider extends SprykerPriceProductStorageDependencyProvider
	{
		/**
		 * @return \Spryker\Client\PriceProductStorageExtension\Dependency\Plugin\PriceProductStoragePricesExtractorPluginInterface[]
		 */
		protected function getPriceProductDecisionPlugins(): array
		{
			return array_merge([                        
				...
				new PriceProductVolumeFilterPlugin(),
			], parent::getPriceProductDecisionPlugins());
		}
	}
   
	}
  4. Add plugins to Zed's PriceProductDependencyProvider
    ModulePluginDescriptionMethod in Dependency Provider
    PriceProductVolumePriceProductVolumeExtractorPluginExtracts volume prices from database price data.getPriceProductPricesExtractorPlugins()
    
	<?php
	namespace Pyz\Zed\PriceProduct;
 
	use Spryker\Zed\PriceProductVolume\Communication\Plugin\PriceProductExtension\PriceProductVolumeExtractorPlugin;
 
	class PriceProductDependencyProvider extends SprykerPriceProductDependencyProvider
	{
		/**
		 * @return \Spryker\Zed\PriceProductExtension\Dependency\Plugin\PriceProductReaderPricesExtractorPluginInterface[]
		 */
		protected function getPriceProductPricesExtractorPlugins(): array
		{
			return [
				new PriceProductVolumeExtractorPlugin(),
			];
		}
	}
  5. Add plugins to Yves's ProductDetailPageDependencyProvider
    ModulePluginDescriptionMethod in Dependency Provider
    VolumePriceProductWidgetVolumePriceProductWidgetPluginRenders volume prices table.getProductDetailPageWidgetPlugins()
    
	<?php
	namespace Pyz\Yves\ProductDetailPage;
 
	use SprykerShop\Yves\ProductDetailPage\ProductDetailPageDependencyProvider as SprykerShopProductDetailPageDependencyProvider;
	use SprykerShop\Yves\PriceProductVolumeWidget\Plugin\ProductDetailPage\PriceProductVolumeWidgetPlugin;
 
	class ProductDetailPageDependencyProvider extends SprykerShopProductDetailPageDependencyProvider
	{
		/**
		 * @return \Spryker\Zed\CompanyUnitAddressGuiExtension\Dependency\Plugin\CompanyUnitAddressEditFormExpanderPluginInterface[]
		 */
		protected function getProductDetailPageWidgetPlugins(): array
		{
			return [
				...
				VolumePriceProductWidgetPlugin::class, #VolumePriceProductFeature
			];
  
		}
 
	}
  6. Run the following commands
    
	console transfer:generate
	console propel:install
	console data:import:product-price

That's it! You can use Volume Pricing feature now!
